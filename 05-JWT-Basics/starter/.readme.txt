require('dotenv').config();
require('express-async-errors');

const express = require('express');
const app = express();

const mainRouter = require('./routes/main'); // Assuming this is where your main router is defined
const notFoundMiddleware = require('./middleware/not-found');
const errorHandlerMiddleware = require('./middleware/error-handler');

// Middleware
app.use(express.static('./public'));
app.use(express.urlencoded({ extended: true })); // Middleware to parse URL-encoded form data

// Routes
app.use('/api/v1', mainRouter); // Assuming mainRouter handles login/authentication

// Error handling middleware
app.use(notFoundMiddleware);
app.use(errorHandlerMiddleware);

const port = process.env.PORT || 3000;

const start = async () => {
  try {
    app.listen(port, () =>
      console.log(`Server is listening on port ${port}...`)
    );
  } catch (error) {
    console.log(error);
  }
};

start();

- The express.urlencoded({ extended: true }) middleware is added to parse URL-encoded form data, which will be used for handling username and password submissions.
- The mainRouter is used to handle the login/authentication logic. This router should handle POST requests for submitting username and password for authentication.
